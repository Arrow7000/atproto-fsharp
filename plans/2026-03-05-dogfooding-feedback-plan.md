```fsharp
let login (baseUrl : string) (identifier : string) (password : string) : Task<Result<AtpAgent, XrpcError>> =
    task {
        let agent = AtpAgent.create baseUrl
        let! result = AtpAgent.login identifier password agent
        match result with
        | Error e -> return Error e
        | Ok _ ->
            // Resolve PDS endpoint from DID document
            match agent.Session with
            | Some session ->
                let! identityResult = Identity.resolveDid agent session.Did
                match identityResult with
                | Ok identity ->
                    match identity.PdsEndpoint with
                    | Some pdsUrl -> agent.BaseUrl <- pdsUrl
                    | None -> () // Keep original URL if no PDS in DID doc
                | Error _ -> () // Keep original URL if resolution fails
            | None -> ()
            return Ok agent
    }
```

Apply the same pattern to `loginWithClient` and `resumeSession`.

### Task 3: Fix withChatProxy session sharing (Bug B3)

`withChatProxy` creates a record copy, but since `Session` is mutable, the copy's `Session` field is independent. The `Chat` module already calls `ensureChatProxy` internally, so the real fix from Bug B2 (PDS resolution) should resolve most issues. But we should also fix the session sharing so `withChatProxy` works correctly regardless of call order.

**Files:**
- Modify: `src/FSharp.ATProto.Core/AtpAgent.fs:82-84` (withChatProxy)
- Test: `tests/FSharp.ATProto.Bluesky.Tests/BlueskyTests.fs`

**Step 2: Run to verify it fails**

Expected: FAIL because the record copy has independent `Session`.

**Step 3: Fix by documenting the constraint and ensuring Chat module handles it**

The `Chat` module already uses `ensureChatProxy` (Chat.fs:29) which adds the header on-the-fly from the original agent. So `withChatProxy` before login already works through the Chat module. The issue is only if someone uses `withChatProxy` directly.

The simplest fix: the `Chat.ensureChatProxy` function should propagate the session from the source. Actually, it already does because it only adds a header to the existing agent or creates a copy with current session state.

Re-examine the bug: the real fix is Bug B2 (PDS resolution makes the entryway URL a non-issue) plus ensuring `Chat.ensureChatProxy` is used consistently (it already is). Document that `withChatProxy` is not needed for the high-level `Chat.*` functions.

If we still want to share session state, we'd need to change `AtpAgent` from a record to a reference type or use a shared `ref`. This is a bigger refactor. **Recommendation: close this as fixed-by-B2 + documented, don't refactor AtpAgent.**

**Step 4: Add PostEmbed mapping module**

```fsharp
module PostEmbed =

    let private mapImage (vi : AppBskyEmbed.Images.ViewImage) : PostImage =
        { Thumb = vi.Thumb; Fullsize = vi.Fullsize; Alt = vi.Alt }

    let private mapVideo (vv : AppBskyEmbed.Video.View) : PostVideo =
        { Thumbnail = vv.Thumbnail; Playlist = vv.Playlist; Alt = vv.Alt }

    let private mapExternal (ve : AppBskyEmbed.External.ViewExternal) : PostExternalLink =
        { Uri = ve.Uri; Title = ve.Title; Description = ve.Description; Thumb = ve.Thumb }

    let ofEmbedUnion (eu : AppBskyFeed.Defs.PostViewEmbedUnion) : PostEmbed =
        match eu with
        | AppBskyFeed.Defs.PostViewEmbedUnion.View iv ->
            PostEmbed.Images (iv.Images |> List.map mapImage)
        | AppBskyFeed.Defs.PostViewEmbedUnion.View2 vv ->
            PostEmbed.Video (mapVideo vv)
        | AppBskyFeed.Defs.PostViewEmbedUnion.View3 ev ->
            PostEmbed.ExternalLink (mapExternal ev.External)
        | AppBskyFeed.Defs.PostViewEmbedUnion.View4 rv ->
            PostEmbed.QuotedPost rv.Record
        | AppBskyFeed.Defs.PostViewEmbedUnion.View5 rwm ->
            let record = rwm.Record.Record
            let media =
                match rwm.Media with
                | AppBskyEmbed.RecordWithMedia.ViewMediaUnion.View iv ->
                    PostMediaEmbed.Images (iv.Images |> List.map mapImage)
                | AppBskyEmbed.RecordWithMedia.ViewMediaUnion.View2 vv ->
                    PostMediaEmbed.Video (mapVideo vv)
                | _ -> PostMediaEmbed.Images [] // fallback
            PostEmbed.RecordWithMedia (record, media)
        | _ -> PostEmbed.Unknown
```

```fsharp
module Notification =

    let private extractText (record : JsonElement) : string =
        match record.TryGetProperty("text") with
        | true, v when v.ValueKind = JsonValueKind.String -> v.GetString()
        | _ -> ""

    let private extractSubjectCid (record : JsonElement) : Cid option =
        match record.TryGetProperty("subject") with
        | true, subj ->
            match subj.TryGetProperty("cid") with
            | true, cidEl when cidEl.ValueKind = JsonValueKind.String ->
                Cid.parse (cidEl.GetString()) |> Result.toOption
            | _ -> None
        | _ -> None

    let private makePostRef (subjectUri : AtUri option) (record : JsonElement) : PostRef =
        let uri = subjectUri |> Option.defaultWith (fun () ->
            AtUri.parse "at://unknown/unknown/unknown" |> Result.defaultWith failwith)
        let cid = extractSubjectCid record |> Option.defaultWith (fun () ->
            Cid.parse "bafyreie5737gdxlw5i64a" |> Result.defaultWith failwith)
        { PostRef.Uri = uri; Cid = cid }

    let ofRaw (n : AppBskyNotification.ListNotifications.Notification) : Notification =
        let content =
            match n.Reason with
            | AppBskyNotification.ListNotifications.NotificationReason.Like ->
                NotificationContent.Like (makePostRef n.ReasonSubject n.Record)
            | AppBskyNotification.ListNotifications.NotificationReason.Repost ->
                NotificationContent.Repost (makePostRef n.ReasonSubject n.Record)
            | AppBskyNotification.ListNotifications.NotificationReason.Follow ->
                NotificationContent.Follow
            | AppBskyNotification.ListNotifications.NotificationReason.Reply ->
                NotificationContent.Reply (extractText n.Record, makePostRef n.ReasonSubject n.Record)
            | AppBskyNotification.ListNotifications.NotificationReason.Mention ->
                NotificationContent.Mention (extractText n.Record)
            | AppBskyNotification.ListNotifications.NotificationReason.Quote ->
                NotificationContent.Quote (extractText n.Record, makePostRef n.ReasonSubject n.Record)
            | AppBskyNotification.ListNotifications.NotificationReason.StarterpackJoined ->

                | None -> NotificationContent.Unknown "starterpack-joined"

                NotificationContent.Unknown s
            | other -> NotificationContent.Unknown (string other)
```

```fsharp
let private setProfileField (agent : AtpAgent) (update : AppBskyActor.Profile.Profile -> AppBskyActor.Profile.Profile) : Task<Result<unit, XrpcError>> =
    taskResult {
        let! (profile, cid) = getProfileRecord agent
        let updated = update profile
        let! putResult = putProfileRecord agent updated cid |> Task.map Some
        match putResult with
        | Some (Ok ()) -> return ()
        | Some (Error e) when e.Error = Some "InvalidSwap" ->
            // Retry once: re-read and re-apply
            let! (profile2, cid2) = getProfileRecord agent
            do! putProfileRecord agent (update profile2) cid2
        | Some (Error e) -> return! Task.FromResult (Error e)
        | None -> return ()
    }

let setDisplayName (agent : AtpAgent) (name : string option) : Task<Result<unit, XrpcError>> =
    setProfileField agent (fun p -> { p with DisplayName = name })

let setDescription (agent : AtpAgent) (description : string option) : Task<Result<unit, XrpcError>> =
    setProfileField agent (fun p -> { p with Description = description })

let setAvatar (agent : AtpAgent) (avatar : (byte[] * ImageMime) option) : Task<Result<unit, XrpcError>> =
    taskResult {
        let! blobJson =
            match avatar with
            | None -> Task.FromResult (Ok None)
            | Some (data, mime) ->
                uploadBlob agent data mime |> Task.map (Result.map (fun br -> Some br.Json))
        do! setProfileField agent (fun p -> { p with Avatar = blobJson })
    }
```
